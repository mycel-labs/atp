42/// The maximum number of stable memory pages a canister can address.
43pub const MAX_PAGES: u64 = u64::MAX / WASM_PAGE_SIZE;

45pub trait Memory {
46    /// Returns the current size of the stable memory in WebAssembly
47    /// pages. (One WebAssembly page is 64Ki bytes.)
48    fn size(&self) -> u64;
49
50    /// Tries to grow the memory by `pages` many pages containing
51    /// zeroes.  If successful, returns the previous size of the
52    /// memory (in pages).  Otherwise, returns -1.
53    fn grow(&self, pages: u64) -> i64;
54
55    /// Copies the data referred to by `offset` out of the stable memory
56    /// and replaces the corresponding bytes in `dst`.
57    fn read(&self, offset: u64, dst: &mut [u8]);
58
59    /// Copies `count` number of bytes of the data starting from `offset` out of the stable memory
60    /// into the buffer starting at `dst`.
61    ///
62    /// This method is an alternative to `read` which does not require initializing a buffer and may
63    /// therefore be faster.
64    ///
65    /// # Safety
66    ///
67    /// Callers must guarantee that
68    ///   * it is valid to write `count` number of bytes starting from `dst`,
69    ///   * `dst..dst + count` does not overlap with `self`.
70    ///
71    /// Implementations must guarantee that before the method returns, `count` number of bytes
72    /// starting from `dst` will be initialized.
73    #[inline]
74    unsafe fn read_unsafe(&self, offset: u64, dst: *mut u8, count: usize) {
75        // Initialize the buffer to make the slice valid.
76        std::ptr::write_bytes(dst, 0, count);
77        let slice = std::slice::from_raw_parts_mut(dst, count);
78        self.read(offset, slice)
79    }
80
81    /// Copies the data referred to by `src` and replaces the
82    /// corresponding segment starting at offset in the stable memory.
83    fn write(&self, offset: u64, src: &[u8]);
84}
85
86/// Copies `count` bytes of data starting from `addr` out of the stable memory into `dst`.
87///
88/// Callers are allowed to pass vectors in any state (e.g. empty vectors).
89/// After the method returns, `dst.len() == count`.
90/// This method is an alternative to `read` which does not require initializing a buffer and may
91/// therefore be faster.
92#[inline]
93fn read_to_vec<M: Memory>(m: &M, addr: Address, dst: &mut std::vec::Vec<u8>, count: usize) {
94    dst.clear();
95    dst.reserve_exact(count);
96    unsafe {
97        m.read_unsafe(addr.get(), dst.as_mut_ptr(), count);
98        // SAFETY: read_unsafe guarantees to initialize the first `count` bytes
99        dst.set_len(count);
100    }
101}

103/// A helper function that reads a single 32bit integer encoded as
104/// little-endian from the specified memory at the specified offset.
105fn read_u32<M: Memory>(m: &M, addr: Address) -> u32 {
106    let mut buf: [u8; 4] = [0; 4];
107    m.read(addr.get(), &mut buf);
108    u32::from_le_bytes(buf)
109}

111/// A helper function that reads a single 64bit integer encoded as
112/// little-endian from the specified memory at the specified offset.
113fn read_u64<M: Memory>(m: &M, addr: Address) -> u64 {
114    let mut buf: [u8; 8] = [0; 8];
115    m.read(addr.get(), &mut buf);
116    u64::from_le_bytes(buf)
117}

119/// Writes a single 32-bit integer encoded as little-endian.
120fn write_u32<M: Memory>(m: &M, addr: Address, val: u32) {
121    write(m, addr.get(), &val.to_le_bytes());
122}

129#[derive(Debug, PartialEq, Eq)]
130pub struct GrowFailed {
131    current_size: u64,
132    delta: u64,
133}
134
135impl Display for GrowFailed {
136    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
137        write!(
138            f,
139            "Failed to grow memory: current size={}, delta={}",
140            self.current_size, self.delta
141        )
142    }
143}
144
145impl error::Error for GrowFailed {}
146
147/// Writes the bytes at the specified offset, growing the memory size if needed.
148fn safe_write<M: Memory>(memory: &M, offset: u64, bytes: &[u8]) -> Result<(), GrowFailed> {
149    let last_byte = offset
150        .checked_add(bytes.len() as u64)
151        .expect("Address space overflow");
152
153    let size_pages = memory.size();
154    let size_bytes = size_pages
155        .checked_mul(WASM_PAGE_SIZE)
156        .expect("Address space overflow");
157
158    if size_bytes < last_byte {
159        let diff_bytes = last_byte - size_bytes;
160        let diff_pages = diff_bytes
161            .checked_add(WASM_PAGE_SIZE - 1)
162            .expect("Address space overflow")
163            / WASM_PAGE_SIZE;
164        if memory.grow(diff_pages) == -1 {
165            return Err(GrowFailed {
166                current_size: size_pages,
167                delta: diff_pages,
168            });
169        }
170    }
171    memory.write(offset, bytes);
172    Ok(())
173}
174
175/// Like [safe_write], but panics if the memory.grow fails.
176fn write<M: Memory>(memory: &M, offset: u64, bytes: &[u8]) {
177    if let Err(GrowFailed {
178        current_size,
179        delta,
180    }) = safe_write(memory, offset, bytes)
181    {
182        panic!(
183            "Failed to grow memory from {} pages to {} pages (delta = {} pages).",
184            current_size,
185            current_size + delta,
186            delta
187        );
188    }
189}

191/// Reads a struct from memory.
192fn read_struct<T, M: Memory>(addr: Address, memory: &M) -> T {
193    let mut value = MaybeUninit::<T>::uninit();
194    unsafe {
195        memory.read_unsafe(
196            addr.get(),
197            value.as_mut_ptr() as *mut u8,
198            core::mem::size_of::<T>(),
199        );
200        value.assume_init()
201    }
202}
203
204/// Writes a struct to memory.
205fn write_struct<T, M: Memory>(t: &T, addr: Address, memory: &M) {
206    let slice = unsafe {
207        core::slice::from_raw_parts(t as *const _ as *const u8, core::mem::size_of::<T>())
208    };
209
210    write(memory, addr.get(), slice)
211}

213/// RestrictedMemory creates a limited view of another memory.  This
214/// allows one to divide the main memory into non-intersecting ranges
215/// and use different layouts in each region.
216#[derive(Clone)]
217pub struct RestrictedMemory<M: Memory> {
218    page_range: core::ops::Range<u64>,
219    memory: M,
220}
221
222impl<M: Memory> RestrictedMemory<M> {
223    pub fn new(memory: M, page_range: core::ops::Range<u64>) -> Self {
224        assert!(page_range.end <= MAX_PAGES);
225        Self { memory, page_range }
226    }
227}
228
229impl<M: Memory> Memory for RestrictedMemory<M> {
230    fn size(&self) -> u64 {
231        let base_size = self.memory.size();
232        if base_size < self.page_range.start {
233            0
234        } else if base_size > self.page_range.end {
235            self.page_range.end - self.page_range.start
236        } else {
237            base_size - self.page_range.start
238        }
239    }
240
241    fn grow(&self, delta: u64) -> i64 {
242        let base_size = self.memory.size();
243        if base_size < self.page_range.start {
244            self.memory
245                .grow(self.page_range.start - base_size + delta)
246                .min(0)
247        } else if base_size >= self.page_range.end {
248            if delta == 0 {
249                (self.page_range.end - self.page_range.start) as i64
250            } else {
251                -1
252            }
253        } else {
254            let pages_left = self.page_range.end - base_size;
255            if pages_left < delta {
256                -1
257            } else {
258                let r = self.memory.grow(delta);
259                if r < 0 {
260                    r
261                } else {
262                    r - self.page_range.start as i64
263                }
264            }
265        }
266    }
267
268    fn read(&self, offset: u64, dst: &mut [u8]) {
269        self.memory
270            .read(self.page_range.start * WASM_PAGE_SIZE + offset, dst)
271    }
272
273    unsafe fn read_unsafe(&self, offset: u64, dst: *mut u8, count: usize) {
274        self.memory
275            .read_unsafe(self.page_range.start * WASM_PAGE_SIZE + offset, dst, count)
276    }
277
278    fn write(&self, offset: u64, src: &[u8]) {
279        self.memory
280            .write(self.page_range.start * WASM_PAGE_SIZE + offset, src)
281    }
282}
