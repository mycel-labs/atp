19// The minimum degree to use in the btree.
20// This constant is taken from Rust's std implementation of BTreeMap.
21const B: usize = 6;
22// The maximum number of entries per node.
23const CAPACITY: usize = 2 * B - 1;
24const LAYOUT_VERSION_1: u8 = 1;
25const LAYOUT_VERSION_2: u8 = 2;
26const MAGIC: &[u8; 3] = b"BTN";
27const LEAF_NODE_TYPE: u8 = 0;
28const INTERNAL_NODE_TYPE: u8 = 1;
29// The size of u32 in bytes.
30const U32_SIZE: Bytes = Bytes::new(4);

36    Internal,

40pub type EntryRef<'a, K> = (&'a K, &'a [u8]);

42/// A node of a B-Tree.
43///
44/// There are two versions of a `Node`:
45///
46/// 1. `V1`, which supports only bounded types.
47/// 2. `V2`, which supports both bounded and unbounded types.
48///
49/// See `v1.rs` and `v2.rs` for more details.
50#[derive(Debug)]
51pub struct Node<K: Storable + Ord + Clone> {
52    address: Address,
53    // List of tuples consisting of a key and the encoded value.
54    // INVARIANT: the list is sorted by key.
55    keys_and_encoded_values: Vec<(K, Value)>,
56    // For the key at position I, children[I] points to the left
57    // child of this key and children[I + 1] points to the right child.
58    children: Vec<Address>,
59    node_type: NodeType,
60    version: Version,
61
62    // The address of the overflow page.
63    // In V2, a node can span multiple pages if it exceeds a certain size.
64    overflows: Vec<Address>,
65}
66
67impl<K: Storable + Ord + Clone> Node<K> {
68    /// Loads a node from memory at the given address.
69    pub fn load<M: Memory>(address: Address, page_size: PageSize, memory: &M) -> Self {
70        // Load the header to determine which version the node is, then load the node accordingly.
71        let header: NodeHeader = read_struct(address, memory);
72        assert_eq!(&header.magic, MAGIC, "Bad magic.");
73        match header.version {
74            LAYOUT_VERSION_1 => match page_size {
75                PageSize::Derived(DerivedPageSize {
76                    max_key_size,
77                    max_value_size,
78                }) => Self::load_v1(header, address, max_key_size, max_value_size, memory),
79                PageSize::Value(_) => {
80                    unreachable!("Tried to load a V1 node without a derived PageSize.")
81                }
82            },
83            LAYOUT_VERSION_2 => Self::load_v2(address, page_size, header, memory),
84            unknown_version => unreachable!("Unsupported version {unknown_version}."),
85        }
86    }
87
88    /// Saves the node to memory.
89    pub fn save<M: Memory>(&mut self, allocator: &mut Allocator<M>) {
90        match self.version {
91            Version::V1(_) => self.save_v1(allocator.memory()),
92            Version::V2(_) => self.save_v2(allocator),
93        }
94    }

105    /// Returns the entry with the max key in the subtree.
106    pub fn get_max<M: Memory>(&self, memory: &M) -> Entry<K> {
107        match self.node_type {
108            NodeType::Leaf => {
109                let last_idx = self.keys_and_encoded_values.len() - 1;
110                (
111                    self.keys_and_encoded_values
112                        .last()
113                        .expect("A node can never be empty")
114                        .0
115                        .clone(),
116                    self.value(last_idx, memory).to_vec(),
117                )
118            }
119            NodeType::Internal => {
120                let last_child = Self::load(
121                    *self
122                        .children
123                        .last()
124                        .expect("An internal node must have children."),
125                    self.version.page_size(),
126                    memory,
127                );
128                last_child.get_max(memory)
129            }
130        }
131    }
132
133    /// Returns the entry with min key in the subtree.
134    pub fn get_min<M: Memory>(&self, memory: &M) -> Entry<K> {
135        match self.node_type {
136            NodeType::Leaf => {
137                // NOTE: a node can never be empty, so this access is safe.
138                let entry = self.entry(0, memory);
139                (entry.0.clone(), entry.1.to_vec())
140            }
141            NodeType::Internal => {
142                let first_child = Self::load(
143                    // NOTE: an internal node must have children, so this access is safe.
144                    self.children[0],
145                    self.version.page_size(),
146                    memory,
147                );
148                first_child.get_min(memory)
149            }
150        }
151    }

154    pub fn is_full(&self) -> bool {
155        self.keys_and_encoded_values.len() >= CAPACITY
156    }

158    /// Swaps the entry at index `idx` with the given entry, returning the old entry.
159    pub fn swap_entry<M: Memory>(
160        &mut self,
161        idx: usize,
162        (key, value): Entry<K>,
163        memory: &M,
164    ) -> Entry<K> {
165        let (old_key, old_value) = core::mem::replace(
166            &mut self.keys_and_encoded_values[idx],
167            (key, Value::by_value(value)),
168        );
169        (old_key, self.extract_value(old_value, memory))
170    }

172    /// Returns a reference to the entry at the specified index.
173    pub fn entry<M: Memory>(&self, idx: usize, memory: &M) -> EntryRef<K> {
174        (
175            &self.keys_and_encoded_values[idx].0,
176            self.value(idx, memory),
177        )
178    }

180    /// Returns a reference to the encoded value at the specified index.
181    pub fn value<M: Memory>(&self, idx: usize, memory: &M) -> &[u8] {
182        // Load and cache the value from the underlying memory if needed.
183        self.keys_and_encoded_values[idx]
184            .1
185            .get_or_load(|offset| self.load_value_from_memory(offset, memory))
186    }

188    /// Extracts the contents of value (by loading it first if it's not loaded yet).
189    fn extract_value<M: Memory>(&self, value: Value, memory: &M) -> Vec<u8> {
190        value.take_or_load(|offset| self.load_value_from_memory(offset, memory))
191    }

193    /// Loads a value from stable memory at the given offset of this node.
194    fn load_value_from_memory<M: Memory>(&self, offset: Bytes, memory: &M) -> Vec<u8> {
195        let reader = NodeReader {
196            address: self.address,
197            overflows: &self.overflows,
198            page_size: self.page_size(),
199            memory,
200        };
201
202        let value_len = read_u32(&reader, Address::from(offset.get())) as usize;
203        let mut bytes = vec![];
204        read_to_vec(
205            &reader,
206            Address::from((offset + U32_SIZE).get()),
207            &mut bytes,
208            value_len,
209        );
210
211        bytes
212    }

278    /// Removes an entry from the back of the node.
279    pub fn pop_entry<M: Memory>(&mut self, memory: &M) -> Option<Entry<K>> {
280        let len = self.entries_len();
281        if len == 0 {
282            return None;
283        }
284
285        let (key, last_value) = self
286            .keys_and_encoded_values
287            .pop()
288            .expect("node must not be empty");
289
290        Some((key, self.extract_value(last_value, memory)))
291    }

293    /// Merges the entries and children of the `source` node into self, along with the median entry.
294    ///
295    /// PRECONDITION:
296    ///   * `self` is not empty.
297    ///   * `source` is not empty.
298    ///   * `self` and `source` are of the same node type.
299    ///
300    /// POSTCONDITION:
301    ///   * `source` is deallocated.
302    ///   * all the entries of `source`, as well as the median, are merged into `self`, in sorted
303    ///      order.
304    pub fn merge<M: Memory>(
305        &mut self,
306        mut source: Node<K>,
307        median: Entry<K>,
308        allocator: &mut Allocator<M>,
309    ) {
310        // Load all the values from the source node first, as they will be moved out.
311        for i in 0..source.entries_len() {
312            source.value(i, allocator.memory());
313        }
314
315        if source.key(0) > self.key(0) {
316            // The source node has keys that are greater than self.
317            // Append the source node into self.
318            Self::append(self, &mut source, median);
319        } else {
320            // self has keys that are greater than the source node.
321            // Append self into the source node (which more efficient).
322            Self::append(&mut source, self, median);
323
324            // Move the entries and children into self.
325            self.keys_and_encoded_values = core::mem::take(&mut source.keys_and_encoded_values);
326            self.children = core::mem::take(&mut source.children);
327        }
328
329        self.save(allocator);
330        source.deallocate(allocator);
331    }
332
333    // Appends the entries and children of node `b` into `a`, along with the median entry.
334    //
335    // PRECONDITION:
336    //   * `a` is not empty.
337    //   * `b` is not empty.
338    //   * `a` and `b` are of the same node type.
339    //   * keys of `a` < median < keys of `b`
340    //
341    // POSTCONDITION:
342    //   * `b` is empty.
343    fn append(a: &mut Node<K>, b: &mut Node<K>, median: Entry<K>) {
344        // Assert preconditions.
345        let a_len = a.entries_len();
346        assert_eq!(a.node_type(), b.node_type());
347        assert!(b.entries_len() > 0);
348        assert!(a_len > 0);
349        assert!(a.key(a_len - 1) < &median.0);
350        assert!(&median.0 < b.key(0));
351
352        a.push_entry(median);
353
354        a.keys_and_encoded_values
355            .append(&mut b.keys_and_encoded_values);
356
357        // Move the children (if any exist).
358        a.children.append(&mut b.children);
359
360        // Assert postconditions.
361        assert_eq!(b.keys_and_encoded_values.len(), 0);
362        assert_eq!(b.children.len(), 0);
363    }

393    ///
394    /// If the key is found then `Result::Ok` is returned, containing the index

421    /// Moves elements from own node to a sibling node and returns the median element.
422    pub fn split<M: Memory>(&mut self, sibling: &mut Node<K>, memory: &M) -> Entry<K> {
423        debug_assert!(self.is_full());
424
425        // Load the values that will be moved out of the node and into the new sibling.
426        for idx in B..self.entries_len() {
427            self.value(idx, memory);
428        }
429
430        // Move the entries and children above the median into the new sibling.
431        sibling.keys_and_encoded_values = self.keys_and_encoded_values.split_off(B);
432        if self.node_type == NodeType::Internal {
433            sibling.children = self.children.split_off(B);
434        }
435
436        // Return the median entry.
437        self.pop_entry(memory)
438            .expect("An initially full node cannot be empty")
439    }

441    /// Deallocates a node.
442    pub fn deallocate<M: Memory>(self, allocator: &mut Allocator<M>) {
443        for overflow in self.overflows.into_iter() {
444            allocator.deallocate(overflow);
445        }
446
447        allocator.deallocate(self.address);
448    }

466/// The value in a K/V pair.
467#[derive(Debug)]
468enum Value {
469    /// The value's encoded bytes.
470    ByVal(Vec<u8>),
471
472    ByRef {
473        /// The value's offset in the node.
474        offset: Bytes,
475        /// The lazily loaded encoded bytes.
476        loaded_value: OnceCell<Vec<u8>>,
477    },
478}
479
480impl Value {
481    pub fn by_ref(offset: Bytes) -> Self {
482        Self::ByRef {
483            offset,
484            loaded_value: Default::default(),
485        }
486    }
487
488    pub fn by_value(value: Vec<u8>) -> Self {
489        Self::ByVal(value)
490    }
491
492    /// Returns a reference to the value if the value has been loaded or runs the given function to
493    /// load the value.
494    pub fn get_or_load(&self, load: impl FnOnce(Bytes) -> Vec<u8>) -> &[u8] {
495        match self {
496            Value::ByVal(v) => &v[..],
497            Value::ByRef {
498                offset,
499                loaded_value: value,
500            } => value.get_or_init(|| load(*offset)),
501        }
502    }
503
504    /// Extracts the value while consuming self if the value has been loaded or runs the given
505    /// function to load the value.
506    pub fn take_or_load(self, load: impl FnOnce(Bytes) -> Vec<u8>) -> Vec<u8> {
507        match self {
508            Value::ByVal(v) => v,
509            Value::ByRef {
510                offset,
511                loaded_value: value,
512            } => value.into_inner().unwrap_or_else(|| load(offset)),
513        }
514    }
515}

517/// Stores version-specific data.
518#[derive(Debug, PartialEq, Copy, Clone, Eq)]
519pub enum Version {
520    /// V1 nodes have a page size derived from the max key/value sizes.
521    V1(DerivedPageSize),
522    /// V2 nodes have a fixed page size.
523    V2(PageSize),
524}
525
526impl Version {
527    pub fn page_size(&self) -> PageSize {
528        match self {
529            Self::V1(page_size) => PageSize::Derived(*page_size),
530            Self::V2(page_size) => *page_size,
531        }
532    }
533}

535/// The size of an individual page in the memory where nodes are stored.
536/// A node, if it's bigger than a single page, overflows into multiple pages.
537#[allow(dead_code)]
538#[derive(Debug, PartialEq, Copy, Clone, Eq)]
539pub enum PageSize {
540    /// Derived page sizes are used when migrating nodes from v1 to v2.
541    /// A migration from v1 nodes to v2 is done incrementally. Children of a v2 node
542    /// may be a v1 node, and storing the maximum sizes around is necessary to be able
543    /// to load v1 nodes.
544    Derived(DerivedPageSize),
545    Value(u32),
546}
547
548impl PageSize {
549    pub fn get(&self) -> u32 {
550        match self {
551            Self::Value(page_size) => *page_size,
552            Self::Derived(page_size) => page_size.get(),
553        }
554    }
555}
556
557/// A page size derived from the maximum sizes of the keys and values.
558#[derive(Debug, PartialEq, Copy, Clone, Eq)]
559pub struct DerivedPageSize {
560    pub max_key_size: u32,
561    pub max_value_size: u32,
562}
563
564impl DerivedPageSize {
565    // Returns the page size derived from the max key/value sizes.
566    fn get(&self) -> u32 {
567        v1::size_v1(self.max_key_size, self.max_value_size).get() as u32
568    }
