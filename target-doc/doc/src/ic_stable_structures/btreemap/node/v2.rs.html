1//! Node V2
2//!
3//! A v2 node is an iteration on a v1 node. Compared to a v1 node, it has a
4//! smaller memory footprint and support for unbounded types.
5//!
6//! # Memory Layout
7//!
8//! To support unbounded types, a v2 node is variable in size and can span multiple
9//! pages of memory if needed [^note]. There are two types of pages:
10//!
11//! 1. Initial Page (the first page of the node)
12//! 2. Overflow page (subsequent pages of the node)
13//!
14//! ## Initial Page Memory Layout
15//!
16//! ```text
17//! ---------------------------------------- <-- Header
18//! Magic "BTN"             ↕ 3 bytes
19//! ----------------------------------------
20//! Layout version (2)      ↕ 1 byte
21//! ----------------------------------------
22//! Node type               ↕ 1 byte
23//! ----------------------------------------
24//! # Entries (k)           ↕ 2 bytes
25//! ----------------------------------------
26//! Overflow address        ↕ 8 bytes
27//! ---------------------------------------- <-- Children (Address 15)
28//! Child(0) address        ↕ 8 bytes
29//! ----------------------------------------
30//! ...
31//! ----------------------------------------
32//! Child(k + 1) address    ↕ 8 bytes
33//! ---------------------------------------- <-- Keys
34//! Key(0)
35//! ----------------------------------------
36//! ...
37//! ----------------------------------------
38//! Key(k)
39//! ---------------------------------------- <-- Values
40//! Value(0)
41//! ----------------------------------------
42//! ...
43//! ----------------------------------------
44//! Value(k)
45//! ----------------------------------------
46//! ```
47//!
48//! ## Overflow Page Memory Layout
49//!
50//! If the data to be stored in the initial page layout is larger than the page size,
51//! then overflow pages are used.
52//!
53//! ```text
54//! ----------------------------------------
55//! Magic "NOF"             ↕ 3 bytes
56//! ----------------------------------------
57//! Next address            ↕ 8 byte
58//! ----------------------------------------
59//! Data
60//! ----------------------------------------
61//! ```
62//!
63//! ## Keys and Values
64//! Keys and values are both encoded in memory as blobs.
65//!
66//! If they are variable in size (i.e. their `IS_FIXED` attribute is set to false),
67//! then the size of the blob is encoded before the blob itself. Otherwise, no size
68//! information is stored.
69//!
70//! [^note]: The page here refers to a fixed-size chunk of memory that is provided to the
71//! node by the BTreeMap Allocator, and has no connection with OS memory pages or
72//! Wasm pages.
73

75use crate::btreemap::Allocator;

78// Initial page
79pub(super) const OVERFLOW_ADDRESS_OFFSET: Bytes = Bytes::new(7);
80const ENTRIES_OFFSET: Bytes = Bytes::new(15);
81
82// Overflow page
83pub(super) const OVERFLOW_MAGIC: &[u8; 3] = b"NOF";
84pub(super) const PAGE_OVERFLOW_NEXT_OFFSET: Bytes = Bytes::new(3);
85pub(super) const PAGE_OVERFLOW_DATA_OFFSET: Bytes = Bytes::new(11);
86
87// The minimum size a page can have.
88// Rationale: a page size needs to at least store the header (15 bytes) + all the children
89// addresses (88 bytes). We round that up to 128 to get a nice binary number.
90const MINIMUM_PAGE_SIZE: u32 = 128;
91
92impl<K: Storable + Ord + Clone> Node<K> {
93    /// Creates a new v2 node at the given address.
94    pub fn new_v2(address: Address, node_type: NodeType, page_size: PageSize) -> Node<K> {
95        assert!(page_size.get() >= MINIMUM_PAGE_SIZE);
96
97        Node {
98            address,
99            node_type,
100            version: Version::V2(page_size),
101            keys_and_encoded_values: vec![],
102            children: vec![],
103            overflows: Vec::with_capacity(0),
104        }
105    }
106
107    /// Loads a v2 node from memory at the given address.
108    pub(super) fn load_v2<M: Memory>(
109        address: Address,
110        page_size: PageSize,
111        header: NodeHeader,
112        memory: &M,
113    ) -> Self {
114        #[cfg(feature = "canbench")]
115        let _p = canbench::profile("node_load_v2");
116
117        // Load the node, including any overflows, into a buffer.
118        let overflows = read_overflows(address, memory);
119
120        let reader = NodeReader {
121            address,
122            overflows: &overflows,
123            page_size,
124            memory,
125        };
126
127        let mut offset = Address::from(0);
128
129        // Load the node type.
130        let node_type = match header.node_type {
131            LEAF_NODE_TYPE => NodeType::Leaf,
132            INTERNAL_NODE_TYPE => NodeType::Internal,
133            other => unreachable!("Unknown node type {}", other),
134        };
135
136        // Load the number of entries
137        let num_entries = header.num_entries as usize;
138
139        // Load children if this is an internal node.
140        offset += ENTRIES_OFFSET;
141        let mut children = vec![];
142        if node_type == NodeType::Internal {
143            // The number of children is equal to the number of entries + 1.
144            children.reserve_exact(num_entries + 1);
145            for _ in 0..num_entries + 1 {
146                let child = Address::from(read_u64(&reader, offset));
147                offset += Address::size();
148                children.push(child);
149            }
150        }
151
152        // Load the keys.
153        let mut keys_encoded_values = Vec::with_capacity(num_entries);
154        let mut buf = vec![];
155        for _ in 0..num_entries {
156            // Load the key's size.
157            let key_size = if K::BOUND.is_fixed_size() {
158                // Key is fixed in size. The size of the key is always its max size.
159                K::BOUND.max_size()
160            } else {
161                // Key is not fixed in size. Read the size from memory.
162                let value = read_u32(&reader, offset);
163                offset += U32_SIZE;
164                value
165            };
166
167            // Load the key.
168            read_to_vec(&reader, offset, &mut buf, key_size as usize);
169            let key = K::from_bytes(Cow::Borrowed(&buf));
170            offset += Bytes::from(key_size);
171            keys_encoded_values.push((key, Value::by_ref(Bytes::from(0usize))));
172        }
173
174        // Load the values
175        for (_key, value) in keys_encoded_values.iter_mut() {
176            // Load the values lazily.
177            *value = Value::by_ref(Bytes::from(offset.get()));
178            let value_size = read_u32(&reader, offset) as usize;
179            offset += U32_SIZE + Bytes::from(value_size as u64);
180        }
181
182        Self {
183            address,
184            keys_and_encoded_values: keys_encoded_values,
185            children,
186            node_type,
187            version: Version::V2(page_size),
188            overflows,
189        }
190    }
191
192    // Saves the node to memory.
193    pub(super) fn save_v2<M: Memory>(&mut self, allocator: &mut Allocator<M>) {
194        #[cfg(feature = "canbench")]
195        let _p = canbench::profile("node_save_v2");
196
197        let page_size = self.version.page_size().get();
198        assert!(page_size >= MINIMUM_PAGE_SIZE);
199
200        // Load all the values. This is necessary so that we don't overwrite referenced
201        // values when writing the entries to the node.
202        for i in 0..self.keys_and_encoded_values.len() {
203            self.value(i, allocator.memory());
204        }
205
206        // Initialize a NodeWriter. The NodeWriter takes care of allocating/deallocating
207        // overflow pages as needed.
208        let mut writer = NodeWriter::new(
209            self.address,
210            self.overflows.clone(),
211            self.page_size(),
212            allocator,
213        );
214
215        // Write the header of the initial page.
216        let mut offset = Address::from(0);
217        let header = NodeHeader {
218            magic: *MAGIC,
219            version: LAYOUT_VERSION_2,
220            node_type: match self.node_type {
221                NodeType::Leaf => LEAF_NODE_TYPE,
222                NodeType::Internal => INTERNAL_NODE_TYPE,
223            },
224            num_entries: self.keys_and_encoded_values.len() as u16,
225        };
226
227        writer.write_struct(&header, offset);
228        offset += NodeHeader::size();
229
230        // Add a null overflow address.
231        // This might get overwritten later in case the node does overflow.
232        writer.write_u64(offset, self.overflows.first().unwrap_or(&NULL).get());
233        offset += Bytes::from(8u64);
234
235        // Write the children
236        for child in self.children.iter() {
237            writer.write_u64(offset, child.get());
238            offset += Address::size();
239        }
240
241        // Write the keys.
242        for (key, _) in self.keys_and_encoded_values.iter() {
243            let key_bytes = key.to_bytes_checked();
244
245            // Write the size of the key if it isn't fixed in size.
246            if !K::BOUND.is_fixed_size() {
247                writer.write_u32(offset, key_bytes.len() as u32);
248                offset += U32_SIZE;
249            }
250
251            // Write the key.
252            writer.write(offset, key_bytes.borrow());
253            offset += Bytes::from(key_bytes.len());
254        }
255
256        // Write the values.
257        for idx in 0..self.entries_len() {
258            // Write the size of the value.
259            let value = self.value(idx, writer.memory());
260            writer.write_u32(offset, value.len() as u32);
261            offset += U32_SIZE;
262
263            // Write the value.
264            writer.write(offset, value);
265            offset += Bytes::from(value.len());
266        }
267
268        self.overflows = writer.finish();
269    }
270}

272fn read_overflows<M: Memory>(address: Address, memory: &M) -> Vec<Address> {
273    #[repr(C, packed)]
274    struct OverflowPageHeader {
275        magic: [u8; 3],
276        next: Address,
277    }
278
279    let mut overflows = vec![];
280    let mut overflow = Address::from(read_u64(memory, address + OVERFLOW_ADDRESS_OFFSET));
281    while overflow != NULL {
282        overflows.push(overflow);
283
284        let header: OverflowPageHeader = read_struct(overflow, memory);
285        assert_eq!(&header.magic, OVERFLOW_MAGIC, "Bad overflow magic.");
286        overflow = header.next;
287    }
288
289    overflows
290}
