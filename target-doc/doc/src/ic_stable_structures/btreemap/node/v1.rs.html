1//! Node V1
2//!
3//! A v1 node is the first node layout, with fixed node sizes and support for
4//! bounded types only.
5//!
6//! # Memory Layout
7//!
8//! ```text
9//! ---------------------------------------- <-- Header
10//! Magic "BTN"             ↕ 3 bytes
11//! ----------------------------------------
12//! Layout version (2)      ↕ 1 byte
13//! ----------------------------------------
14//! Node type               ↕ 1 byte
15//! ----------------------------------------
16//! # Entries (k)           ↕ 2 bytes
17//! ---------------------------------------- <-- Entries (upto `CAPACITY` entries)
18//! Key(0)
19//! ----------------------------------------
20//! Value(0)
21//! ----------------------------------------
22//! Key(1) size             ↕ 4 bytes
23//! ----------------------------------------
24//! Key(1)                  ↕ `max_key_size` bytes
25//! ----------------------------------------
26//! Value(1) size           ↕ 4 bytes
27//! ----------------------------------------
28//! Value(1)                ↕ `max_value_size` bytes
29//! ----------------------------------------
30//! ...
31//! ---------------------------------------- <-- Children (upto `CAPACITY + 1` children)
32//! Child(0) address        ↕ 8 bytes
33//! ----------------------------------------
34//! ...
35//! ----------------------------------------
36//! Child(k + 1) address    ↕ 8 bytes
37//! ----------------------------------------
38//! ```
39use super::*;

41impl<K: Storable + Ord + Clone> Node<K> {
42    /// Creates a new v1 node at the given address.
43    pub fn new_v1(address: Address, node_type: NodeType, page_size: DerivedPageSize) -> Node<K> {
44        Node {
45            address,
46            node_type,
47            keys_and_encoded_values: vec![],
48            children: vec![],
49            version: Version::V1(page_size),
50            overflows: Vec::with_capacity(0),
51        }
52    }
53
54    /// Loads a v1 node from memory at the given address.
55    pub(super) fn load_v1<M: Memory>(
56        header: NodeHeader,
57        address: Address,
58        max_key_size: u32,
59        max_value_size: u32,
60        memory: &M,
61    ) -> Self {
62        #[cfg(feature = "canbench")]
63        let _p = canbench::profile("node_load_v1");
64
65        // Load the entries.
66        let mut keys_encoded_values = Vec::with_capacity(header.num_entries as usize);
67        let mut offset = NodeHeader::size();
68        let mut buf = vec![];
69        for _ in 0..header.num_entries {
70            // Read the key's size.
71            let key_size = read_u32(memory, address + offset);
72            offset += U32_SIZE;
73
74            // Read the key.
75            read_to_vec(memory, address + offset, &mut buf, key_size as usize);
76            offset += Bytes::from(max_key_size);
77            let key = K::from_bytes(Cow::Borrowed(&buf));
78            // Values are loaded lazily. Store a reference and skip loading it.
79            keys_encoded_values.push((key, Value::by_ref(offset)));
80
81            offset += U32_SIZE + Bytes::from(max_value_size);
82        }
83
84        // Load children if this is an internal node.
85        let mut children = vec![];
86        if header.node_type == INTERNAL_NODE_TYPE {
87            // The number of children is equal to the number of entries + 1.
88            children.reserve_exact(header.num_entries as usize + 1);
89            for _ in 0..header.num_entries + 1 {
90                let child = Address::from(read_u64(memory, address + offset));
91                offset += Address::size();
92                children.push(child);
93            }
94
95            assert_eq!(children.len(), keys_encoded_values.len() + 1);
96        }
97
98        Self {
99            address,
100            keys_and_encoded_values: keys_encoded_values,
101            children,
102            node_type: match header.node_type {
103                LEAF_NODE_TYPE => NodeType::Leaf,
104                INTERNAL_NODE_TYPE => NodeType::Internal,
105                other => unreachable!("Unknown node type {}", other),
106            },
107            version: Version::V1(DerivedPageSize {
108                max_key_size,
109                max_value_size,
110            }),
111            overflows: Vec::with_capacity(0),
112        }
113    }
114
115    pub(super) fn save_v1<M: Memory>(&self, memory: &M) {
116        #[cfg(feature = "canbench")]
117        let _p = canbench::profile("node_save_v1");
118
119        match self.node_type {
120            NodeType::Leaf => {
121                assert!(self.children.is_empty());
122            }
123            NodeType::Internal => {
124                assert_eq!(self.children.len(), self.keys_and_encoded_values.len() + 1);
125            }
126        };
127
128        // We should never be saving an empty node.
129        assert!(!self.keys_and_encoded_values.is_empty() || !self.children.is_empty());
130
131        // Assert entries are sorted in strictly increasing order.
132        assert!(self
133            .keys_and_encoded_values
134            .windows(2)
135            .all(|e| e[0].0 < e[1].0));
136
137        let (max_key_size, max_value_size) = match self.version {
138            Version::V1(DerivedPageSize {
139                max_key_size,
140                max_value_size,
141            }) => (max_key_size, max_value_size),
142            Version::V2 { .. } => unreachable!("cannot save v2 node as v1."),
143        };
144
145        let header = NodeHeader {
146            magic: *MAGIC,
147            version: LAYOUT_VERSION_1,
148            node_type: match self.node_type {
149                NodeType::Leaf => LEAF_NODE_TYPE,
150                NodeType::Internal => INTERNAL_NODE_TYPE,
151            },
152            num_entries: self.keys_and_encoded_values.len() as u16,
153        };
154
155        write_struct(&header, self.address, memory);
156
157        let mut offset = NodeHeader::size();
158
159        // Load all the values. This is necessary so that we don't overwrite referenced
160        // values when writing the entries to the node.
161        for i in 0..self.keys_and_encoded_values.len() {
162            self.value(i, memory);
163        }
164
165        // Write the entries.
166        for (idx, (key, _)) in self.keys_and_encoded_values.iter().enumerate() {
167            // Write the size of the key.
168            let key_bytes = key.to_bytes_checked();
169            write_u32(memory, self.address + offset, key_bytes.len() as u32);
170            offset += U32_SIZE;
171
172            // Write the key.
173            write(memory, (self.address + offset).get(), key_bytes.borrow());
174            offset += Bytes::from(max_key_size);
175
176            // Write the size of the value.
177            let value = self.value(idx, memory);
178            write_u32(memory, self.address + offset, value.len() as u32);
179            offset += U32_SIZE;
180
181            // Write the value.
182            write(memory, (self.address + offset).get(), value);
183            offset += Bytes::from(max_value_size);
184        }
185
186        // Write the children
187        for child in self.children.iter() {
188            write(
189                memory,
190                (self.address + offset).get(),
191                &child.get().to_le_bytes(),
192            );
193            offset += Address::size();
194        }
195    }
196}

198/// Returns the size of a v1 node in bytes.
199pub(super) fn size_v1(max_key_size: u32, max_value_size: u32) -> Bytes {
200    let node_header_size = NodeHeader::size();
201    let max_key_size = Bytes::from(max_key_size);
202    let max_value_size = Bytes::from(max_value_size);
203    let entry_size = U32_SIZE + max_key_size + max_value_size + U32_SIZE;
204    let child_size = Address::size();
205
206    node_header_size
207        + Bytes::from(CAPACITY as u64) * entry_size
208        + Bytes::from((CAPACITY + 1) as u64) * child_size
209}
